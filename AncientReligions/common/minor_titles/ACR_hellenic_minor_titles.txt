	monthly_piety = 0.100

	allowed_to_hold = {
		is_female = yes
		is_landed = no
		is_betrothed = no
		is_married = no
		is_consort = no
		prisoner = no
		religion = hellenic_pagan
		OR = {
			has_minor_title = title_vestalis
			AND = {
				age = 6
				NOT = { age = 10 }
				NOT = { trait = bastard }
				NOT = { trait = legit_bastard }
				NOT = { trait = child_of_consort }
				father_even_if_dead = { religion = hellenic_pagan }
				mother = { religion = hellenic_pagan }
			}
		}
	}
		
	allowed_to_grant = {
		has_landed_title = d_hellenic_pagan
	}

	gain_effect = {
		remove_trait = hedonist
		add_trait = celibate
		add_trait = vestalis
		recalc_succession = yes
		set_character_flag = flag_vestalis
		d_hellenic_pagan = { owner = { change_variable = { which = "num_vestales" value = 1 } } }
	}
	lose_effect = {
	}

	message = yes
}
