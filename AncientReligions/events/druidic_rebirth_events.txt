# Old "Wizard" Appears
character_event = {
	id = ancrel.0000
	desc = EVTDESCancrel.0000
	picture = GFX_evt_Whispering_in_ear
	border = GFX_event_normal_frame_religion
	
	ai = no
	min_age = 16
	capable_only = yes
	prisoner = no
	
	trigger = {
		has_dlc = "The Old Gods"
		is_ruler = yes
		in_command = no
		culture_group = celtic
		NOT = {
			OR = {
				religion = druidic
				religion = druidic_reformed
				religion = gaelic_pagan
				religion = gaelic_pagan_reformed
			}
			has_character_flag = hates_magic
			any_courtier = {
				has_minor_title = title_pagan_wizard
			}
		}
	}
	
	mean_time_to_happen = {
		months = 24
	}	
	
	option = {
		name = EVTOPTAancrel.0000
		if = {
			limit = {
				OR = {
					culture =  breton
					culture = welsh
				}
			}
			create_character = {
				random_traits = no
				dynasty = NONE
				female = no
				age = 53
				religion = druidic
				culture = ROOT
				trait = mystic
				trait = scholarly_theologian
			}
		}
		if = {
			limit = {
				OR = {
					culture =  scottish
					culture = irish
				}
			}
			create_character = {
				random_traits = no
				dynasty = NONE
				female = no
				age = 53
				religion = gaelic_pagan
				culture = ROOT
				trait = mystic
				trait = scholarly_theologian
			}
		}
		new_character = {
			give_minor_title = title_pagan_wizard
		}
	}
	option = {
		name = EVTOPTBancrel.0000
		custom_tooltip = {
			text = begone_sorcerer
			set_character_flag = hates_magic
		}
	}
}

# Bishop asks character to reform Celtic ways
character_event = {
	id = ancrel.0002
	desc = EVTDESCancrel.0002
	picture = GFX_evt_scandal
	
	prisoner = no
	capable_only = yes
	
	trigger = {
		NOT = {
			OR = {
				religion = druidic
				religion = gaelic_pagan
			}
		}
		has_character_modifier = celtic_ways
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = EVTOPTAancrel.0002
		ai_chance = { factor = 50 }
		hidden_tooltip = { remove_character_modifier = celtic_ways }
		if = {
			limit = { 
				OR = {
					culture = scottish
					culture = irish
				}
			}
			religion = gaelic_pagan
		}
		if = {
			limit = { 
				OR = {
					culture = welsh
					culture = breton
				}
			}
			religion = druidic
		}
		piety = 50
	}
	option = {
		name = EVTOPTBancrel.0002
		piety = 50
		ai_chance = { factor = 50 }
		religion_head = {
			opinion = {
				modifier = opinion_happy
				who = ROOT
				years = 2
			}
		}
		hidden_tooltip = { remove_character_modifier = celtic_ways }
	}
}

		}

			}

					limit = { is_ruler = yes }
					top_liege = { character_event = { id = ancrel.0004 days = 5 } }
				}
			}
		}
	}
	option = {
		name = EVTOPTBancrel.0003
		piety = 50
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				trait = zealous
			}
		}
		random_vassal = {
			limit = {
				is_theocracy = yes
				religion = catholic
			}
			opinion = {
				modifier = opinion_happy
				who = ROOT
				years = 2
			}
		}
		set_character_flag = refuse_celtic_ways
		hidden_tooltip = {
			IF = {
				limit = { is_ruler = yes }
				top_liege = { character_event = { id = ancrel.0005 days = 5 } }
			}
		}
	}
}
