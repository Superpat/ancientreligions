	hide_window = yes
	
	ai = yes
	min_age = 16
	max_age = 40
	only_men = yes
	capable_only = yes
	prisoner = no
	
	trigger = {
		has_dlc = "The Old Gods"
		is_landed = no
		in_command = no
		has_global_flag = fianna_founded
		OR = {
			religion = druidic
			religion = druidic_reformed
		}
		OR = {
			culture = irish
			culture = scottish
		}
		NOT = {
			trait = fian 
			has_character_flag = wants_fianna
		}
		father = {
			is_alive = yes
			OR = {
				religion = druidic
				religion = druidic_reformed
			}
			OR = {
				culture = irish
				culture = scottish
			}
			is_ruler = yes
			is_liege_of = ROOT
			prisoner = no
			NOT = { trait = incapable }
			ROOT = { host = { character = PREVPREV } }
		}
	}
	
	mean_time_to_happen = {
		months = 240
		
		modifier = {
			factor = 0.5
			trait = ambitious
		}
		modifier = {
			factor = 0.5
			trait = brave
		}
		modifier = {
			factor = 0.75
			OR = {
				trait = head1
				trait = head2
				trait = head3
				trait = head4
				trait = head5
			}
		}
		modifier = {
			factor = 0.75
			trait = proud
		}
		modifier = {
			factor = 0.75
			trait = impaler
		}
		modifier = {
			factor = 1.5
			trait = slothful
		}
		modifier = {
			factor = 1.5
			trait = content
		}
		modifier = {
			factor = 1.5
			is_primary_heir = yes
		}
		modifier = {
			factor = 2.0
			trait = craven
		}
	}
	
	immediate = {
		father = { character_event = { id = ancrel.1160 } }
		set_character_flag = wants_fianna
	}
	
	option = {
		name = OK
	}
}
