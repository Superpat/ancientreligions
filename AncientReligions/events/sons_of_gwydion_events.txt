			religion = druidic_reformed
		}
		NOT = { 
			trait = wolf 
			has_character_flag = wants_gwydion
		}
		father = {
			is_alive = yes
			culture_group = celtic
			is_ruler = yes
			is_liege_of = ROOT

			NOT = { trait = incapable }
			ROOT = { host = { character = PREVPREV } }
		}
	}
	
	mean_time_to_happen = {
		months = 240
		
		modifier = {
			factor = 0.5
			trait = ambitious
		}
		modifier = {
			factor = 0.5
			trait = brave
		}
		modifier = {
			factor = 0.75
			OR = {
				trait = head1
				trait = head2
				trait = head3
				trait = head4
				trait = head5
			}
		}
		modifier = {
			factor = 0.75
			trait = proud
		}
		modifier = {
			factor = 0.75
			trait = impaler
		}
		modifier = {
			factor = 1.5
			trait = slothful
		}
		modifier = {
			factor = 1.5
			trait = content
		}
		modifier = {
			factor = 1.5
			is_primary_heir = yes
		}
		modifier = {
			factor = 2.0
			trait = craven
		}
	}
	
	immediate = {
		father = { character_event = { id = ancrel.0120 } }
		set_character_flag = wants_gwydion
	}
	
	option = {
		name = OK
	}

	id = ancrel.0131
	desc = EVTDESCancrel.0131
	picture = GFX_evt_viking_throneroom_oldgods
	border = GFX_event_normal_frame_war
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTAancrel.0131
		if = {
			limit = {
				FROM = {
					trait = skilled_tactician
				}
			}
			FROM = {
				hidden_tooltip = { remove_trait = skilled_tactician }
				add_trait = brilliant_strategist
			}
		}
		if = {
			limit = {
				FROM = {
					OR = {
						trait = misguided_warrior
						trait = tough_soldier
					}	
				}
			}
			FROM = {
				hidden_tooltip = { 
					remove_trait = misguided_warrior
					remove_trait = tough_soldier
				}
				add_trait = skilled_tactician
			}
		}
		FROM = {
			change_martial = 2
			add_trait = wolf
			move_character = ROOT
		}
	}
}
